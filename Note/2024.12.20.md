class Solution {

    string removeDuplicates(string s) {
        stack<char> middle;
        for(char c:s)
        {
            if(middle.empty())
            {
                middle.push(c);
                continue;
            }
            if(c==middle.top())
            {
                middle.pop();
                continue;
            }
            middle.push(c);

        string res="";

        {
            res+=middle.top();
            middle.pop();
        }

    }
};
