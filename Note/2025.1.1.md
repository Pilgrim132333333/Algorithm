class Solution {
private:
    vector<vector<int>> res;
    vector<int> mid;
    void backtracking(int n,int k,int middle)
    {
        if(mid.size()==k)
        {
            res.push_back(mid);
            return;
        }
        for(int i=middle;i<=n;i++)
        {
            mid.push_back(i);
            backtracking(n,k,i+1);
            mid.pop_back();
        }
    }
public: 
    vector<vector<int>> combine(int n, int k) {
        backtracking(n,k,1);
        return res;
        
    }
};
